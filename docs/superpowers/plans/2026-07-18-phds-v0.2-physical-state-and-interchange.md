# PHDS v0.2 Physical State and Interchange Implementation Plan

> **Status:** Completed implementation record. The commands, checkboxes, and
> task sequencing below document the implementation process; they are not
> current instructions for modifying PHDS.

**Goal:** Deliver PHDS v0.2 with sparse event-time physical state, temporal property addresses, system-qualified ratings, rent rolls, evidence artifacts, deeper provenance, optional standards profiles, and cross-generator conformance.

**Architecture:** Keep the dependency chain `core -> entities -> profiles -> capture`. Stable `Property`, `Site`, `Structure`, and `Space` identities reuse fact mixins with independently identified dated state assertions; optional standards import PHDS but PHDS core never imports them. LinkML remains the source of truth, generated contracts remain committed artifacts, and semantic invariants that LinkML cannot express are enforced by a small profile validator.

**Tech Stack:** LinkML 1.11.1, YAML, JSON Schema Draft 2020-12, Pydantic 2, Python `unittest`, TypeScript generation, Rust/Serde, JSON-LD, `just`.

## Global Constraints

- Never create or amend a git commit without the user's express permission for that specific commit action. Every commit step below is an approval checkpoint, not standing authorization.
- Target schema version is exactly `0.2.0`.
- This is a clean pre-1.0 revision: no deprecated aliases, compatibility fields, or dual representations.
- Core PHDS remains internationally neutral and does not normatively adopt a national, regional, vendor, or asset-class-specific rating standard.
- Core, entities, profiles, and capture must not import anything under `schema/standards/`.
- Stable physical entities and dated state assertions must always have different identifiers.
- Historical state is sparse; absent values remain absent and are never copied from current state.
- `*_pct` fields and `cap_rate` use 0–100 percentage points; `Provenance.confidence` uses a 0–1 fraction.
- Preserve the existing exact decimal-string wire contract for `Money.amount`.
- All schema changes require regenerated JSON Schema, Pydantic, TypeScript, and Rust artifacts.
- Every task ends green before the next task begins.

from __future__ import annotations

import json
from pathlib import Path
import subprocess
import tempfile
import unittest

from linkml.generators.jsonschemagen import JsonSchemaGenerator
from linkml.generators.pydanticgen import PydanticGenerator

from tools.gen_typescript import EnumAwareTypescriptGenerator


ROOT = Path(__file__).resolve().parents[1]

OPTIONAL_SCHEMA = ROOT / "tests/fixtures/linkml/optional_rating_patterns.yaml"


class PhysicalStatePatternTests(unittest.TestCase):
    def test_mixins_flatten_and_state_identity_survives_every_generator(self):
        json_schema = json.loads(JsonSchemaGenerator(str(SCHEMA)).serialize())
        for class_name in ("Structure", "StructureState"):
            properties = json_schema["$defs"][class_name]["properties"]
            self.assertIn("year_built", properties)
            self.assertIn("condition", properties)
            self.assertIn("id", properties)
        self.assertIn("property", json_schema["$defs"]["Structure"]["properties"])
        self.assertIn("subject", json_schema["$defs"]["StructureState"]["properties"])

        pydantic_source = PydanticGenerator(str(SCHEMA)).serialize()
        typescript_source = EnumAwareTypescriptGenerator(str(SCHEMA)).serialize()
        self.assertIn("class Structure(", pydantic_source)
        self.assertIn("class StructureState(", pydantic_source)
        self.assertIn("export interface Structure", typescript_source)
        self.assertIn("export interface StructureState", typescript_source)

        with tempfile.TemporaryDirectory() as tmp:
            subprocess.run(
                [
                    str(ROOT / ".venv/bin/gen-rust"),
                    str(SCHEMA),
                    "--output",
                    tmp,
                    "--force",
                    "--serde",
                ],
                check=True,
                capture_output=True,
                text=True,

            rust_source = (Path(tmp) / "src/lib.rs").read_text(encoding="utf-8")
        self.assertIn("pub struct Structure", rust_source)
        self.assertIn("pub struct StructureState", rust_source)
        self.assertIn("pub subject: String", rust_source)

    def test_optional_import_constrains_without_entering_base_closure(self):
        base = json.loads(JsonSchemaGenerator(str(SCHEMA)).serialize())

        self.assertNotIn("OptionalStructureState", base["$defs"])
        self.assertNotIn("OptionalConditionCode", base["$defs"])

        self.assertEqual("#/$defs/OptionalConditionCode", condition["$ref"])

        pydantic_source = PydanticGenerator(str(OPTIONAL_SCHEMA)).serialize()
        typescript_source = EnumAwareTypescriptGenerator(str(OPTIONAL_SCHEMA)).serialize()
        self.assertIn("class OptionalStructureState(", pydantic_source)
        self.assertIn("condition: OptionalConditionCode", typescript_source)

        with tempfile.TemporaryDirectory() as tmp:
            subprocess.run(
                [
                    str(ROOT / ".venv/bin/gen-rust"),
                    str(OPTIONAL_SCHEMA),
                    "--output",
                    tmp,
                    "--force",
                    "--serde",
                ],
                check=True,
                capture_output=True,
                text=True,
            )
            rust_source = (Path(tmp) / "src/lib.rs").read_text(encoding="utf-8")
        self.assertIn("pub struct OptionalStructureState", rust_source)
        self.assertIn("pub condition: OptionalConditionCode", rust_source)


if __name__ == "__main__":
    unittest.main()

```python
from __future__ import annotations

import argparse
from dataclasses import dataclass
from datetime import date
import json
from pathlib import Path
import sys
from typing import Any


@dataclass(frozen=True, order=True)
class ValidationIssue:
    path: str
    message: str


def _date(value: str | None) -> date | None:
    return date.fromisoformat(value) if value else None


def _role_code(value: Any) -> str:
    if isinstance(value, dict):
        return str(value.get("code", ""))
    return str(value or "")


def _overlaps(left: dict[str, Any], right: dict[str, Any]) -> bool:
    left_start, left_end = _date(left.get("valid_from")), _date(left.get("valid_to"))
    right_start, right_end = _date(right.get("valid_from")), _date(right.get("valid_to"))
    return (left_end is None or right_start is None or right_start <= left_end) and (
        right_end is None or left_start is None or left_start <= right_end
    )


def validate_profile(profile: dict[str, Any]) -> list[ValidationIssue]:
    issues: list[ValidationIssue] = []
    property_id = profile.get("property", {}).get("id")
    address_ids = {item.get("id") for item in profile.get("addresses", [])}

    for index, address in enumerate(profile.get("addresses", [])):
        has_hash = bool(address.get("address_hash"))
        has_scheme = bool(address.get("address_hash_scheme"))
        if has_hash != has_scheme:
            issues.append(ValidationIssue(
                f"addresses[{index}]",
                "address_hash requires address_hash_scheme",
            ))

    primaries: dict[str, list[tuple[int, dict[str, Any]]]] = {}
    for index, association in enumerate(profile.get("property_addresses", [])):
        path = f"property_addresses[{index}]"
        if association.get("property") != property_id:
            issues.append(ValidationIssue(path, "property reference does not match profile property"))
        if association.get("address") not in address_ids:
            issues.append(ValidationIssue(path, "address reference does not resolve"))
        if (
            association.get("valid_from")
            and association.get("valid_to")
            and _date(association["valid_to"]) < _date(association["valid_from"])
        ):
            issues.append(ValidationIssue(path, "valid_to precedes valid_from"))
        if association.get("is_primary"):
            primaries.setdefault(_role_code(association.get("role")), []).append(
                (index, association)
            )

    for role, values in primaries.items():
        for left_index, left in values:
            for right_index, right in values:
                if left_index < right_index and _overlaps(left, right):
                    issues.append(ValidationIssue(
                        f"property_addresses[{right_index}]",
                        f"primary address validity overlaps for role {role}",
                    ))

    stable_ids = {
        "property_state": {property_id},
        "site_states": {profile.get("site", {}).get("id")} if profile.get("site") else set(),
        "structure_states": {item.get("id") for item in profile.get("structures", [])},
        "space_states": {item.get("id") for item in profile.get("spaces", [])},
    }
    snapshot_ids = {item.get("id") for item in profile.get("property_state_snapshots", [])}
    for snapshot_index, snapshot in enumerate(profile.get("property_state_snapshots", [])):
        snapshot_path = f"property_state_snapshots[{snapshot_index}]"
        if snapshot.get("property") != property_id:
            issues.append(ValidationIssue(snapshot_path, "snapshot property does not match profile property"))
        groups = {
            "property_state": [snapshot["property_state"]] if snapshot.get("property_state") else [],
            "site_states": snapshot.get("site_states", []),
            "structure_states": snapshot.get("structure_states", []),
            "space_states": snapshot.get("space_states", []),
        }
        for group, states in groups.items():
            for state_index, state in enumerate(states):
                if state.get("subject") not in stable_ids[group]:
                    issues.append(ValidationIssue(
                        f"{snapshot_path}.{group}[{state_index}]",
                        "state subject does not resolve",
                    ))

    for collection in ("sales", "leases", "listings", "valuations"):
        for index, event in enumerate(profile.get(collection, [])):
            reference = event.get("property_state")
            if reference and reference not in snapshot_ids:
                issues.append(ValidationIssue(
                    f"{collection}[{index}].property_state",
                    "property_state reference does not resolve",
                ))

    return sorted(issues)


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("profile", type=Path)
    args = parser.parse_args()
    profile = json.loads(args.profile.read_text(encoding="utf-8"))
    issues = validate_profile(profile)
    for issue in issues:
        print(f"{issue.path}: {issue.message}", file=sys.stderr)
    return 1 if issues else 0


if __name__ == "__main__":
    raise SystemExit(main())
```

def _validate_rent_rolls(
    profile: dict[str, Any], issues: list[ValidationIssue]
) -> None:
    property_id = profile.get("property", {}).get("id")
    space_ids = {item.get("id") for item in profile.get("spaces", [])}
    lease_ids = {item.get("id") for item in profile.get("leases", [])}
    money_fields = ("total_contract_rent", "total_market_rent")
    line_money_fields = ("contract_rent", "market_rent")

    for roll_index, roll in enumerate(profile.get("rent_rolls", [])):
        path = f"rent_rolls[{roll_index}]"
        if roll.get("property") != property_id:
            issues.append(ValidationIssue(path, "rent roll property does not match profile property"))
        if (
            roll.get("unit_count") is not None
            and roll.get("occupied_unit_count") is not None
            and roll["occupied_unit_count"] > roll["unit_count"]
        ):
            issues.append(ValidationIssue(path, "occupied_unit_count exceeds unit_count"))

        currencies: set[str] = set()
        for field in money_fields:
            if isinstance(roll.get(field), dict) and roll[field].get("currency"):
                currencies.add(roll[field]["currency"])
        for line_index, line in enumerate(roll.get("lines", [])):
            line_path = f"{path}.lines[{line_index}]"
            for field in line_money_fields:
                if isinstance(line.get(field), dict) and line[field].get("currency"):

                ("lease", lease_ids),
            ):
                if line.get(field) and line[field] not in valid_ids:
                    issues.append(ValidationIssue(line_path, f"{field} reference does not resolve"))
        if len(currencies) > 1:

```python
def _validate_artifacts(
    profile: dict[str, Any], issues: list[ValidationIssue]
) -> None:
    artifacts = profile.get("artifacts", [])
    artifact_ids = {item.get("id") for item in artifacts}
    for index, artifact in enumerate(artifacts):
        has_hash = bool(artifact.get("content_hash"))
        has_scheme = bool(artifact.get("hash_scheme"))
        if has_hash != has_scheme:
            issues.append(ValidationIssue(
                f"artifacts[{index}]",
                "content_hash requires hash_scheme",
            ))

    def walk(value: Any, path: str) -> None:
        if isinstance(value, dict):
            for key, child in value.items():
                child_path = f"{path}.{key}" if path else key
                if key == "artifacts" and path:
                    for ref_index, reference in enumerate(child):
                        if reference not in artifact_ids:
                            issues.append(ValidationIssue(
                                f"{child_path}[{ref_index}]",
                                "artifact reference does not resolve",
                            ))
                else:
                    walk(child, child_path)
        elif isinstance(value, list):
            for index, child in enumerate(value):
                walk(child, f"{path}[{index}]")

    for key, value in profile.items():
        if key != "artifacts":
            walk(value, key)
```
